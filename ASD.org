  (defun go-to-string (file string)
    (progn
      (find-file file)
      (goto-char (point-min))
      (search-forward string)))

  (defun my/org-expand-defsystem-line ()
    (interactive)
    (let* ((line (thing-at-point 'line t))
           (re "(defsystem \"\\([^\"]+\\)\"")
           name base asd-file link)
      (when (string-match re line)
        (setq name (match-string 1 line))          ;; mnas-sdl3-gui/events
        (setq base (car (split-string name "/")))  ;; mnas-sdl3-gui
        (setq asd-file (format "%s.asd" base))
        (setq link (format "[[elisp:(go-to-string %s \"(defsystem \\\"%s\\\"\")][%s]]"
                           asd-file name name))
        (beginning-of-line)
        (kill-line)
        (insert link))))
